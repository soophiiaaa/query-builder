    protected string $table;
    protected array $where = [];
    protected array $params = [];

    public const AND = 'AND';
    public const OR = 'OR';
    public const EQUALS = '=';
    public const NOT_EQUAL = '!=';
    public const GREATER_THAN = '>';
    public const LESS_THAN = '<';
    public const GREATER_THAN_OR_EQUAL = '>=';
    public const LESS_THAN_OR_EQUAL = '<=';
    public const NOT_LIKE_OR_EQUAL = '<>';

    public function __construct(string $table)
    {
        $this->table = $table;
    }

    public function where(
        string $column,
        string $operator,
        mixed $value,
        string $boolean = self::AND
    )
    {
        $prefix = empty($this->where) ? '' : $boolean . ' ';
        
        $this->where[]  = "{$prefix}{$column} {$operator}  ?";
        $this->params[] = $value;
    }

    public function reset(): void
    {
        $this->where  = [];
        $this->params = [];
    }

    abstract function toSql(): string;

    public function getBindings(): array
    {
        return $this->where;
    }

    public function __tostring(): string
    {
        return $this->toSql();
    }
